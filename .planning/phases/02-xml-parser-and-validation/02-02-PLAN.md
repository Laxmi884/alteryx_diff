Public parser API (src/alteryx_diff/parser.py):
  def parse(path_a: pathlib.Path, path_b: pathlib.Path) -> tuple[WorkflowDoc, WorkflowDoc]

Exception hierarchy (src/alteryx_diff/exceptions.py):
  ParseError(Exception)              -- base; carries .filepath: str and .message: str
  MalformedXMLError(ParseError)      -- lxml XMLSyntaxError (includes empty file)
  MissingFileError(ParseError)       -- path does not exist
  UnreadableFileError(ParseError)    -- path exists, not a regular file (directory, permissions)

WorkflowDoc shape (from alteryx_diff.models):
  WorkflowDoc.filepath: str
  WorkflowDoc.nodes: tuple[AlteryxNode, ...]
  WorkflowDoc.connections: tuple[AlteryxConnection, ...]
  AlteryxNode.tool_id: ToolID (int subtype), .tool_type: str, .x: float, .y: float, .config: dict[str, Any]
  AlteryxConnection.src_tool: ToolID, .src_anchor: AnchorName, .dst_tool: ToolID, .dst_anchor: AnchorName

Config dict convention from _element_to_dict:
  XML attributes -> "@key" entries in dict
  Non-whitespace text -> "#text" key
  Repeated same-tag children -> list value
  Example: <Configuration><File RecordLimit="0">data.csv</File></Configuration>
    -> {"File": {"@RecordLimit": "0", "#text": "data.csv"}}
  Example: <Fields><Field name="id"/><Field name="val"/></Fields>
    -> {"Fields": {"Field": [{"@name": "id"}, {"@name": "val"}]}}

  import pytest

Helper to write a fixture file:
  def write_fixture(tmp_path: pathlib.Path, name: str, content: bytes) -> pathlib.Path:
      p = tmp_path / name
      p.write_bytes(content)
      return p

Tests to implement (one function per test, descriptive names):

test_parse_happy_path(tmp_path):
  Write MINIMAL_YXMD to two temp files. Call parse(path_a, path_b).
  Assert: returns tuple of two WorkflowDoc instances.
  Assert: doc_a.filepath ends with the temp file name.
  Assert: len(doc_a.nodes) == expected tool count from MINIMAL_YXMD fixture.
  Assert: doc_a.nodes[0].tool_id is an int (ToolID is NewType of int).
  Assert: doc_a.nodes[0].x and .y are floats.
  Assert: doc_a.nodes[0].tool_type is a non-empty string.
  Assert: isinstance(doc_a.connections, tuple).

test_parse_config_dict_shape(tmp_path):
  Write MINIMAL_YXMD to a temp file. parse(path, path).
  Assert: doc.nodes[0].config is a dict.
  Assert: config contains expected nested keys matching the fixture's Configuration element.
  Assert: XML attributes appear as "@key" entries in the dict.

test_parse_repeated_config_children(tmp_path):
  Write REPEATED_FIELDS_YXMD to a temp file. parse(path, path).
  Navigate to the Field list in the config dict.
  Assert: the repeated Field children are a list, not a single dict.
  Assert: len(that list) >= 2.

test_parse_empty_workflow(tmp_path):
  Write EMPTY_WORKFLOW_YXMD to a temp file. parse(path, path).
  Assert: doc.nodes == ().
  Assert: doc.connections == ().

test_parse_dual_file_independence(tmp_path):
  Write TWO_NODE_YXMD to path_a and MINIMAL_YXMD to path_b.
  doc_a, doc_b = parse(path_a, path_b).
  Assert: doc_a.filepath != doc_b.filepath.
  Assert: doc_a.nodes are independent from doc_b.nodes (different objects).

test_parse_missing_file_raises(tmp_path):
  path_a = tmp_path / "nonexistent_a.yxmd"  # do NOT create it
  path_b = write_fixture(tmp_path, "valid_b.yxmd", MINIMAL_YXMD)
  with pytest.raises(MissingFileError) as exc_info:
      parse(path_a, path_b)
  assert "nonexistent_a.yxmd" in exc_info.value.filepath
  assert exc_info.value.message  # non-empty plain-English message

test_parse_malformed_xml_raises(tmp_path):
  path_a = write_fixture(tmp_path, "malformed.yxmd", MALFORMED_XML)
  path_b = write_fixture(tmp_path, "valid.yxmd", MINIMAL_YXMD)
  with pytest.raises(MalformedXMLError) as exc_info:
      parse(path_a, path_b)
  assert exc_info.value.filepath  # non-empty
  assert exc_info.value.__cause__ is not None  # __cause__ chained

test_parse_empty_file_raises(tmp_path):
  path_a = write_fixture(tmp_path, "empty.yxmd", EMPTY_FILE)
  path_b = write_fixture(tmp_path, "valid.yxmd", MINIMAL_YXMD)
  with pytest.raises(MalformedXMLError):
      parse(path_a, path_b)

test_parse_binary_content_raises(tmp_path):
  path_a = write_fixture(tmp_path, "binary.yxmd", BINARY_CONTENT)
  path_b = write_fixture(tmp_path, "valid.yxmd", MINIMAL_YXMD)
  with pytest.raises(MalformedXMLError):
      parse(path_a, path_b)

test_parse_directory_raises(tmp_path):
  dir_path = tmp_path / "a_directory"
  dir_path.mkdir()
  path_b = write_fixture(tmp_path, "valid.yxmd", MINIMAL_YXMD)
  with pytest.raises((UnreadableFileError, MalformedXMLError)):
      parse(dir_path, path_b)

test_parse_fail_fast_on_path_a(tmp_path):
  Verify parse() never attempts path_b when path_a fails.
  path_a = tmp_path / "missing.yxmd"  # does not exist
  path_b = tmp_path / "also_missing.yxmd"  # also does not exist
  with pytest.raises(MissingFileError) as exc_info:
      parse(path_a, path_b)
  assert "missing.yxmd" in exc_info.value.filepath
  assert "also_missing.yxmd" not in exc_info.value.filepath  # path_b not mentioned

test_parse_error_is_parse_error_subclass(tmp_path):
  All raised errors must subclass ParseError (for Phase 9 CLI to catch with a single except).
  path_a = tmp_path / "nonexistent.yxmd"
  path_b = write_fixture(tmp_path, "valid.yxmd", MINIMAL_YXMD)
  with pytest.raises(ParseError):
      parse(path_a, path_b)

Do NOT use any mocking library. All tests use real file I/O in tmp_path. No parametrize unless it simplifies a clear pattern.
