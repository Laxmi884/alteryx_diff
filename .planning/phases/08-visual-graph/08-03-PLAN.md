Helper function to extract GRAPH_NODES from fragment string:
```python
def _extract_graph_nodes(html: str) -> list[dict]:
    marker = "var GRAPH_NODES = "
    start = html.index(marker) + len(marker)
    # Find the closing ];
    end = html.index("]", start) + 1
    return json.loads(html[start:end])
```

def test_canvas_layout_uses_alteryx_coordinates() -> None:
    from alteryx_diff.models.types import ToolID, AnchorName
    from alteryx_diff.models.workflow import AlteryxNode
    n_old = AlteryxNode(tool_id=ToolID(820), tool_type="InputData", x=100.0, y=200.0, config={})
